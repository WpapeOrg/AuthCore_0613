cd /Users/wpape/Sourcetree/AuthCore_0613

# 方式一：sqlite3 交互式（macOS 自带）
sqlite3 users.db
sqlite> .tables              # 列出所有表
sqlite> .schema users        # 查看 users 表结构
sqlite> SELECT * FROM users; # 查看所有用户
sqlite> .mode column         # 列对齐显示
sqlite> .headers on          # 显示列名
sqlite> SELECT id, username, email, created_at FROM users;
sqlite> .quit

# 方式二：单条命令直接查询
sqlite3 users.db "SELECT id, username, email FROM users;"

# 方式三：导出为 CSV
sqlite3 -header -csv users.db "SELECT * FROM users;" > users_export.csv


方法	用途	返回值

db.exec(sql)	执行 DDL（建表等），不返回数据	this（链式）
db.prepare(sql).run(a, b)	执行 INSERT/UPDATE/DELETE	{ changes, lastInsertRowid }
db.prepare(sql).get(a)	查询单行	对象 或 undefined
db.prepare(sql).all(a)	查询多行	对象数组
db.prepare(sql).iterate(a)	逐行迭代（内存友好）	Iterator
db.transaction(fn)	创建事务函数	包装后的函数
db.pragma('key = val')	设置/读取 SQLite 编译选项	取决于具体 pragma
db.close()	关闭数据库连接
